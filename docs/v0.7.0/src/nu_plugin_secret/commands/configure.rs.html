1//! Configuration management commands for nu_plugin_secret
2//!
3//! This module provides runtime configuration management commands including:
4//! - `secret configure` - Interactive configuration changes
5//! - `secret config show` - Display current settings  
6//! - `secret config reset` - Restore defaults
7//! - `secret config validate` - Validate configuration
8//! - `secret config export/import` - Configuration backup/restore
9
10use nu_plugin::{EngineInterface, EvaluatedCall, PluginCommand};
11use nu_protocol::{
12    Category, Example, LabeledError, PipelineData, Record, Signature, SyntaxShape, Type, Value,
13};
14
15use crate::config::{ConfigManager, PluginConfig, SecurityLevel};
16
17/// Main configuration command that provides subcommands for config management
18pub struct SecretConfigureCommand;
19
20/// Parse a security level string and apply it to the current config, returning
21/// the updated config ready for validation and persistence.
22fn apply_security_level_change(
23    plugin: &crate::SecretPlugin,
24    level_str: &str,
25    span: nu_protocol::Span,
26) -> Result<PluginConfig, LabeledError> {
27    let mut config = plugin
28        .config_manager()
29        .read()
30        .map_err(|e| {
31            LabeledError::new("Configuration Error")
32                .with_label(format!("Failed to read configuration: {}", e), span)
33        })?
34        .config()
35        .clone();
36
37    let level = match level_str {
38        "minimal" => SecurityLevel::Minimal,
39        "standard" => SecurityLevel::Standard,
40        "paranoid" => SecurityLevel::Paranoid,
41        _ => {
42            return Err(LabeledError::new("Invalid Security Level").with_label(
43                format!(
44                    "Unknown level '{}'. Valid options: minimal, standard, paranoid",
45                    level_str
46                ),
47                span,
48            ));
49        }
50    };
51    config.security.level = level;
52
53    Ok(config)
54}

56/// Build a summary record of the current configuration state for output.
57fn build_config_summary_record(
58    config_manager: &ConfigManager,
59    config_changed: bool,
60    span: nu_protocol::Span,
61) -> Record {
62    let mut record = Record::new();
63
64    record.push(
65        "redaction_template",
66        Value::string(
67            config_manager.config().redaction.get_redaction_template(),
68            span,
69        ),
70    );
71
72    record.push(
73        "security_level",
74        Value::string(
75            format!("{:?}", config_manager.config().security.level).to_lowercase(),
76            span,
77        ),
78    );
79
80    if config_changed {
81        record.push(
82            "status",
83            Value::string("Configuration updated successfully", span),
84        );
85    } else {
86        record.push("status", Value::string("No changes made", span));
87    }
88
89    record
90}

92impl PluginCommand for SecretConfigureCommand {
93    type Plugin = crate::SecretPlugin;
94
95    fn name(&self) -> &str {
96        "secret configure"
97    }
98
99    fn description(&self) -> &str {
100        "Configure secret plugin settings interactively"
101    }
102
103    fn signature(&self) -> Signature {
104        Signature::build(self.name())
105            .input_output_types(vec![(Type::Nothing, Type::Record(Box::new([])))])
106            .named(
107                "security-level",
108                SyntaxShape::String,
109                "Set security level (minimal, standard, paranoid)",
110                Some('s'),
111            )
112            .category(Category::Custom("secret".into()))
113    }
114
115    fn examples(&self) -> Vec<Example<'_>> {
116        vec![Example {
117            example: "secret configure --security-level paranoid",
118            description: "Set security level to paranoid (maximum security)",
119            result: None,
120        }]
121    }
122
123    fn run(
124        &self,
125        plugin: &Self::Plugin,
126        _engine: &EngineInterface,
127        call: &EvaluatedCall,
128        _input: PipelineData,
129    ) -> Result<PipelineData, LabeledError> {
130        let span = call.head;
131
132        let mut config_changed = false;
133        let mut new_config = None;
134
135        // Handle security level changes
136        if let Some(level_str) = call.get_flag::<String>("security-level")? {
137            new_config = Some(apply_security_level_change(plugin, &level_str, span)?);
138            config_changed = true;
139        }
140
141        // Validate and apply configuration changes
142        if let Some(config) = new_config {
143            // Validate configuration
144            if let Err(e) = ConfigManager::validate_config(&config) {
145                return Err(LabeledError::new("Configuration Validation Failed")
146                    .with_label(format!("Invalid configuration: {}", e), span));
147            }
148
149            // Audit the change if enabled
150            if let Ok(current_manager) = plugin.config_manager().read() {
151                if current_manager.config().security.audit_config_changes {
152                    let _ = crate::config::audit_config_change(current_manager.config(), &config);
153                }
154            }
155
156            // Update plugin configuration
157            let mut config_manager = plugin.config_manager().write().map_err(|e| {
158                LabeledError::new("Configuration Error")
159                    .with_label(format!("Failed to acquire write lock: {}", e), span)
160            })?;
161
162            *config_manager.config_mut() = config;
163
164            // Save to disk
165            config_manager.save().map_err(|e| {
166                LabeledError::new("Save Error")
167                    .with_label(format!("Failed to save configuration: {}", e), span)
168            })?;
169        }
170
171        // Create summary record of current configuration
172        let config_manager = plugin.config_manager().read().map_err(|e| {
173            LabeledError::new("Configuration Error")
174                .with_label(format!("Failed to read configuration: {}", e), span)
175        })?;
176
177        let record = build_config_summary_record(&config_manager, config_changed, span);
178
179        Ok(PipelineData::Value(Value::record(record, span), None))
180    }
181}
