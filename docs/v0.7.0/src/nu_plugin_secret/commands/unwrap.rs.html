1//! Implements `secret unwrap` — extracts the underlying value from a secret.
2
3use nu_plugin::{EngineInterface, EvaluatedCall, PluginCommand};
4use nu_protocol::{Category, Example, LabeledError, PipelineData, Signature, Type, Value};
5
6use crate::{
7    SecretBinary, SecretBool, SecretDate, SecretFloat, SecretInt, SecretList, SecretRecord,
8    SecretString,
9};
10
11#[derive(Clone)]
12pub struct SecretUnwrapCommand;
13
14impl PluginCommand for SecretUnwrapCommand {
15    type Plugin = crate::SecretPlugin;
16
17    fn name(&self) -> &str {
18        "secret unwrap"
19    }
20
21    fn signature(&self) -> Signature {
22        Signature::build(self.name())
23            .input_output_types(vec![
24                (Type::Custom("secret_string".into()), Type::String),
25                (Type::Custom("secret_int".into()), Type::Int),
26                (Type::Custom("secret_bool".into()), Type::Bool),
27                (
28                    Type::Custom("secret_record".into()),
29                    Type::Record(Box::new([])),
30                ),
31                (
32                    Type::Custom("secret_list".into()),
33                    Type::List(Box::new(Type::Any)),
34                ),
35                (Type::Custom("secret_float".into()), Type::Float),
36                (Type::Custom("secret_binary".into()), Type::Binary),
37                (Type::Custom("secret_date".into()), Type::Date),
38            ])
39            .category(Category::Conversions)
40    }
41
42    fn description(&self) -> &str {
43        "Extract the underlying value from any secret type. WARNING: This exposes sensitive data!"
44    }
45
46    fn examples(&self) -> Vec<Example<'_>> {
47        vec![Example {
48            example: r#""my-secret" | secret wrap | secret unwrap"#,
49            description: "Unwrap a secret string to get the original value",
50            result: Some(Value::test_string("my-secret")),
51        }]
52    }
53
54    fn run(
55        &self,
56        _plugin: &Self::Plugin,
57        _engine: &EngineInterface,
58        call: &EvaluatedCall,
59        input: PipelineData,
60    ) -> Result<PipelineData, LabeledError> {
61        match input {
62            PipelineData::Value(Value::Custom { val, .. }, metadata) => {
63                // Security warning would be logged here in a real implementation
64                // For now, we'll just proceed with the unwrap operation
65
66                if let Some(secret_string) = val.as_any().downcast_ref::<SecretString>() {
67                    let revealed = secret_string.reveal().to_string();
68                    let value = Value::string(revealed, call.head);
69                    Ok(PipelineData::Value(value, metadata))
70                } else if let Some(secret_int) = val.as_any().downcast_ref::<SecretInt>() {
71                    let revealed = secret_int.reveal();
72                    let value = Value::int(revealed, call.head);
73                    Ok(PipelineData::Value(value, metadata))
74                } else if let Some(secret_bool) = val.as_any().downcast_ref::<SecretBool>() {
75                    let revealed = secret_bool.reveal();
76                    let value = Value::bool(revealed, call.head);
77                    Ok(PipelineData::Value(value, metadata))
78                } else if let Some(secret_record) = val.as_any().downcast_ref::<SecretRecord>() {
79                    let revealed = secret_record.reveal().clone();
80                    let value = Value::record(revealed, call.head);
81                    Ok(PipelineData::Value(value, metadata))
82                } else if let Some(secret_list) = val.as_any().downcast_ref::<SecretList>() {
83                    let revealed = secret_list.reveal().clone();
84                    let value = Value::list(revealed, call.head);
85                    Ok(PipelineData::Value(value, metadata))
86                } else if let Some(secret_float) = val.as_any().downcast_ref::<SecretFloat>() {
87                    let revealed = secret_float.reveal();
88                    let value = Value::float(revealed, call.head);
89                    Ok(PipelineData::Value(value, metadata))
90                } else if let Some(secret_binary) = val.as_any().downcast_ref::<SecretBinary>() {
91                    let revealed = secret_binary.reveal().clone();
92                    let value = Value::binary(revealed, call.head);
93                    Ok(PipelineData::Value(value, metadata))
94                } else if let Some(secret_date) = val.as_any().downcast_ref::<SecretDate>() {
95                    let revealed = *secret_date.reveal();
96                    let value = Value::date(revealed, call.head);
97                    Ok(PipelineData::Value(value, metadata))
98                } else {
99                    Err(LabeledError::new("Type Error")
100                        .with_label("Expected a secret type", call.head)
101                        .with_help("Only secret types can be unwrapped"))
102                }
103            }
104            PipelineData::Value(value, _) => Err(LabeledError::new("Type Error")
105                .with_label(
106                    format!("Expected secret type, got {}", value.get_type()),
107                    call.head,
108                )
109                .with_help("Only secret types can be unwrapped")),
110            PipelineData::Empty => Err(LabeledError::new("Empty Input")
111                .with_label("No input provided", call.head)
112                .with_help("Provide a secret value to unwrap")),
113            _ => Err(LabeledError::new("Unsupported Input")
114                .with_label("Cannot unwrap stream input", call.head)
115                .with_help("Provide a single secret value")),
116        }
117    }
118}
