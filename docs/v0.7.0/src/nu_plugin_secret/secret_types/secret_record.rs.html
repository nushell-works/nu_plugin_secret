15/// A secure record type that redacts its content in all display contexts
16/// and zeros its memory on drop
17#[derive(Clone)]
18pub struct SecretRecord {
19    inner: Record,
20    redaction_template: Option<String>,
21}
22
23// Functional serialization - serialize actual content for pipeline operations
24impl Serialize for SecretRecord {
25    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
26    where
27        S: Serializer,
28    {
29        use serde::ser::SerializeStruct;
30        let mut state = serializer.serialize_struct("SecretRecord", 2)?;
31        state.serialize_field("inner", &self.inner)?;
32        state.serialize_field("redaction_template", &self.redaction_template)?;
33        state.end()
34    }
35}
36
37// Functional deserialization - restore actual content for pipeline operations
38impl<'de> Deserialize<'de> for SecretRecord {
39    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
40    where
41        D: Deserializer<'de>,
42    {
43        #[derive(Deserialize)]
44        struct SecretRecordData {
45            inner: Record,
46            redaction_template: Option<String>,
47        }
48
49        let data = SecretRecordData::deserialize(deserializer)?;
50        Ok(SecretRecord {
51            inner: data.inner,
52            redaction_template: data.redaction_template,
53        })
54    }
55}

57impl Drop for SecretRecord {
58    fn drop(&mut self) {
59        // Note: We rely on ZeroizeOnDrop for additional memory clearing
60        // The Record will be properly dropped by Rust's destructor
61        // Cannot safely zero the Record's memory as it contains complex structures
62    }
63}

65// Manual ZeroizeOnDrop implementation to ensure proper cleanup
66impl ZeroizeOnDrop for SecretRecord {}

68impl SecretRecord {
69    /// Create a new SecretRecord from a regular record
70    pub fn new(value: Record) -> Self {
71        Self {
72            inner: value,
73            redaction_template: None,
74        }
75    }
76
77    /// Create a new SecretRecord with a custom redaction template
78    pub fn new_with_template(value: Record, template: String) -> Self {
79        Self {
80            inner: value,
81            redaction_template: Some(template),
82        }
83    }
84
85    /// Get a reference to the inner record (for controlled access)
86    pub fn reveal(&self) -> &Record {
87        &self.inner
88    }
89
90    /// Convert back to a regular record (consumes the SecretRecord)
91    pub fn into_inner(self) -> Record {
92        self.inner.clone()
93    }
94
95    /// Get a field from the record while preserving secrecy
96    pub fn get_field(&self, field: &str) -> Option<&Value> {
97        self.inner.get(field)
98    }
99
100    /// Check if the record has no fields
101    pub fn is_empty(&self) -> bool {
102        self.inner.is_empty()
103    }
104
105    /// Get all field names (safe to expose)
106    pub fn fields(&self) -> impl Iterator<Item = &String> {
107        self.inner.columns()
108    }
109}

111#[typetag::serde]
112impl CustomValue for SecretRecord {
113    fn clone_value(&self, span: Span) -> Value {
114        Value::custom(Box::new(self.clone()), span)
115    }
116
117    fn type_name(&self) -> String {
118        "secret_record".into()
119    }
120
121    fn to_base_value(&self, span: Span) -> Result<Value, ShellError> {
122        let redacted_text = if let Some(template) = &self.redaction_template {
123            // Convert record to parsable string representation
124            let record_value = Value::record(self.inner.clone(), Span::unknown());
125            let record_str = record_value.to_parsable_string(", ", &nu_protocol::Config::default());
126            crate::redaction::generate_redacted_string_with_custom_template_and_value(
127                template,
128                "record",
129                None,
130                Some(record_str), // Length not meaningful for complex types
131            )
132        } else {
133            crate::redaction::get_redacted_string_with_value::<String>(
134                "record",
135                RedactionContext::Serialization,
136                None,
137            )
138        };
139        Ok(Value::string(redacted_text, span))
140    }
141
142    fn as_any(&self) -> &dyn std::any::Any {
143        self
144    }
145
146    fn as_mut_any(&mut self) -> &mut dyn std::any::Any {
147        self
148    }
149
150    fn notify_plugin_on_drop(&self) -> bool {
151        false // We handle cleanup via ZeroizeOnDrop
152    }
153
154    fn operation(
155        &self,
156        lhs_span: Span,
157        operator: Operator,
158        op: Span,
159        right: &Value,
160    ) -> Result<Value, ShellError> {
161        secret_comparison_operation(self, lhs_span, operator, op, right, "secret_record")
162    }
163}

208impl PartialEq for SecretRecord {
209    fn eq(&self, other: &Self) -> bool {
210        // Compare records field by field for proper logical equality
211        if self.inner.len() != other.inner.len() {
212            return false;
213        }
214
215        // Check that all fields in self exist in other and have equal values
216        for (key, self_value) in self.inner.iter() {
217            match other.inner.get(key) {
218                Some(other_value) => {
219                    if self_value != other_value {
220                        return false;
221                    }
222                }
223                None => return false,
224            }
225        }
226
227        // Since we already checked lengths are equal and all fields in self
228        // exist in other with equal values, the records are equal
229        true
230    }
231}
